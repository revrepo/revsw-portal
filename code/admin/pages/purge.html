{
"version": 1,
"purges": [
  {
    "url": {
      "is_wildcard": true,
      "expression": "/images/*.png"
    }
  }
]
}

{
"version": 1,
"purges": [

    "url": {
      "is_wildcard": true,

  }

Purge everything, recursively, for current domain

      "expression": "/**/*"
    }
  }
]
}

]
